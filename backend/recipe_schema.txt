RECIPES SCHEMA:
CREATE TABLE recipes (
	id INTEGER NOT NULL, 
	source VARCHAR(50) NOT NULL, 
	source_url VARCHAR(500) NOT NULL, 
	source_id VARCHAR(200), 
	name VARCHAR(300) NOT NULL, 
	description TEXT, 
	cuisine VARCHAR(100), 
	category VARCHAR(100), 
	cook_time_min INTEGER, 
	total_time_min INTEGER, 
	difficulty VARCHAR(20), 
	base_servings INTEGER DEFAULT '4' NOT NULL, 
	instructions_json TEXT DEFAULT '[]' NOT NULL, 
	tips_json TEXT DEFAULT '[]' NOT NULL, 
	image_url VARCHAR(500), 
	rating_avg FLOAT, 
	rating_count INTEGER, 
	is_vegetarian BOOLEAN DEFAULT '0' NOT NULL, 
	is_vegan BOOLEAN DEFAULT '0' NOT NULL, 
	is_meat BOOLEAN DEFAULT '0' NOT NULL, 
	is_fish BOOLEAN DEFAULT '0' NOT NULL, 
	contains_pork BOOLEAN DEFAULT '0' NOT NULL, 
	contains_beef BOOLEAN DEFAULT '0' NOT NULL, 
	contains_chicken BOOLEAN DEFAULT '0' NOT NULL, 
	contains_turkey BOOLEAN DEFAULT '0' NOT NULL, 
	allergen_flags_json TEXT DEFAULT '[]' NOT NULL, 
	meat_kinds_json TEXT DEFAULT '[]' NOT NULL, 
	tags_json TEXT DEFAULT '[]' NOT NULL, 
	scraped_at DATETIME NOT NULL, 
	last_seen_at DATETIME NOT NULL, meal_type VARCHAR(30) DEFAULT 'hauptgericht' NOT NULL, 
	PRIMARY KEY (id)
)
